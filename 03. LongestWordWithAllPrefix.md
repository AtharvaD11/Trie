#include <bits/stdc++.h> 
struct Node{
    Node* links[26] = {NULL};
    bool flag = false;

    bool containsChar(char ch){
        return (links[ch-'a'] != NULL);
    }

    void put(char ch, Node* node){
        links[ch-'a'] = node;
    }

    Node *get(char ch) { return links[ch - 'a']; }
    void setEnd() { flag = true; }
};

class Trie{
    private : Node* root;
    public : 
        Trie(){
            root = new Node();
        }

        void insert(string &word){
            Node* node = root;
            for(int i=0;i<word.length();i++){
                if(!node->containsChar(word[i])){
                    Node* refNode = new Node();
                    node->put(word[i], refNode);
                }
                node = node->get(word[i]);
            }
            node->setEnd();
        }

        bool hasAllPrefix(string &word){
            Node* node = root; 
            for(int i=0;i<word.length();i++){
                Node* refNode = node->get(word[i]);
                if(refNode->flag == false){
                    return false;
                }
                node = refNode;
            }
            return true;
        }


};

string completeString(int n, vector<string> &a){
    Trie t;
    for(auto &i : a){
        t.insert(i);
    }
    string str = "ninja";

    string minStr = "";
    for(auto &i : a){
        if(t.hasAllPrefix(i)){
            if(minStr.length() == i.length()){
                minStr= min(minStr, i);
            }
            else if(minStr.length() < i.length()) minStr = i;
        }
    }
    if(minStr.length()) return minStr;
    return "None";
}
